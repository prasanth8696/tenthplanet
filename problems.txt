    @api.depends("status")
    def _compute_color(self):
        if self.status == "new" :
            self.color = 4
        elif self.status == "offer received" :
            self.color = 2
        elif self.status == "offer accepted" :
            self.color = 10
        elif self.status == "canceled" :
            self.color = 9
        else :
            self.color = 3
